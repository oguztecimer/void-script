This component serves as the Console panel header (satisfying PNLS-02 for Console) — it has a title area ("Console" tab label) and right-aligned action icons (clear console, close panel).

```tsx
import { useStore } from '../state/store';
import { ToolBtn } from '../primitives/ToolBtn';
import styles from './BottomTabStrip.module.css';

const BOTTOM_TABS = [
  { id: 'console', label: 'Console' },
];

export function BottomTabStrip() {
  const activeTab = useStore((s) => s.bottomPanelTab);
  const setTab = useStore((s) => s.setBottomPanelTab);
  const clearConsole = useStore((s) => s.clearConsole);
  const toggleBottomPanel = useStore((s) => s.toggleBottomPanel);

  return (
    <div className={styles.strip}>
      <div className={styles.tabs}>
        {BOTTOM_TABS.map((tab) => (
          <button
            key={tab.id}
            className={`${styles.tab} ${activeTab === tab.id ? styles.active : ''}`}
            onClick={() => setTab(tab.id)}
          >
            {tab.label}
          </button>
        ))}
      </div>
      <div className={styles.actions}>
        <ToolBtn size="small" onClick={clearConsole} title="Clear Console">
          <svg width="12" height="12" viewBox="0 0 16 16">
            <path d="M2 2h12M4 6h8M6 10h4M8 14" stroke="currentColor" strokeWidth="1.5"/>
          </svg>
        </ToolBtn>
        <ToolBtn size="small" onClick={toggleBottomPanel} title="Close">
          <svg width="12" height="12" viewBox="0 0 16 16">
            <path d="M4 4l8 8M12 4l-8 8" stroke="currentColor" strokeWidth="1.5"/>
          </svg>
        </ToolBtn>
      </div>
    </div>
  );
}
```

**Update App.tsx layout:**
Import `{ PanelGroup, Panel, PanelResizeHandle }` and `type { ImperativePanelHandle }` from `'react-resizable-panels'`. Also import `{ useRef, useState, useEffect }` from React (add to existing import).

CRITICAL ARCHITECTURE: Panels must ALWAYS be rendered in the tree. Do NOT use `{leftPanelOpen && <Panel>}` — this unmounts the Panel, losing size state and preventing animation. Instead:
1. Always render all three `<Panel>` components
2. Use `useRef<ImperativePanelHandle>` for each side panel
3. Use `useEffect` watching `leftPanelOpen`/`rightPanelOpen` to call `panelRef.current.collapse()` or `panelRef.current.expand()` imperatively
4. Conditionally render the panel CONTENT (not the Panel itself) for panels that should show nothing when "closed" is irrelevant — the Panel collapses to 0% via the imperative API, so the content inside is naturally hidden

For the collapse animation:
1. Add `const [isResizing, setIsResizing] = useState(false)` state
2. Pass `onDragging={setIsResizing}` to BOTH `PanelResizeHandle` components
3. Apply a `className={!isResizing ? styles.panelAnimated : ''}` to each side Panel
4. The `panelAnimated` class adds `transition: flex-basis 150ms ease` — this fires during programmatic collapse/expand but is removed during drag to avoid fighting the pointer

Layout structure:
```tsx
export function App() {
  const leftPanelOpen = useStore((s) => s.leftPanelOpen);
  const rightPanelOpen = useStore((s) => s.rightPanelOpen);
  const bottomPanelOpen = useStore((s) => s.bottomPanelOpen);
  const isDebugging = useStore((s) => s.isDebugging);
  // ... other store selectors ...

  const leftPanelRef = useRef<ImperativePanelHandle>(null);
  const rightPanelRef = useRef<ImperativePanelHandle>(null);
  const [isResizing, setIsResizing] = useState(false);

  // Imperative collapse/expand synced to Zustand state
  useEffect(() => {
    const panel = leftPanelRef.current;
    if (!panel) return;
    if (leftPanelOpen) {
      panel.expand();
    } else {
      panel.collapse();
    }
  }, [leftPanelOpen]);

  useEffect(() => {
    const panel = rightPanelRef.current;
    if (!panel) return;
    if (rightPanelOpen && isDebugging) {
      panel.expand();
    } else {
      panel.collapse();
    }
  }, [rightPanelOpen, isDebugging]);

  return (
    <div className={styles.app}>
      <Header />

      <div className={styles.main}>
        {/* Left tool strip — fixed, OUTSIDE PanelGroup */}
        <ToolStrip side="left" ... />

        {/* Resizable panel layout */}
        <PanelGroup direction="horizontal" autoSaveId="void-main-layout" className={styles.panelGroup}>
          {/* Left panel — ALWAYS rendered, collapse/expand via imperative API */}
          <Panel
            ref={leftPanelRef}
            id="left-panel"
            order={1}
            defaultSize={18}
            minSize={10}
            maxSize={40}
            collapsible
            collapsedSize={0}
            className={!isResizing ? styles.panelAnimated : ''}
          >
            <ScriptList />
          </Panel>

          <PanelResizeHandle
            className={styles.resizeHandle}
            onDragging={setIsResizing}
          />

          {/* Center panel */}
          <Panel id="center" order={2}>
            <div className={styles.center}>
              <TabBar />
              <div className={styles.editorArea}>
                <Editor />
              </div>
              {bottomPanelOpen && (
                <div className={styles.bottomPanel}>
                  <BottomTabStrip />
                  <Console />
                </div>
              )}
            </div>
          </Panel>

          <PanelResizeHandle
            className={styles.resizeHandle}
            onDragging={setIsResizing}
          />

          {/* Right panel — ALWAYS rendered, collapse/expand via imperative API */}
          <Panel
            ref={rightPanelRef}
            id="right-panel"
            order={3}
            defaultSize={18}
            minSize={10}
            maxSize={40}
            collapsible
            collapsedSize={0}
            className={!isResizing ? styles.panelAnimated : ''}
          >
            {isDebugging && <DebugPanel />}
          </Panel>
        </PanelGroup>

        {/* Right tool strip — fixed, OUTSIDE PanelGroup */}
        <ToolStrip side="right" ... />
      </div>

      <StatusBar />
    </div>
  );
}
```

KEY POINTS:
- Both side Panels are ALWAYS in the DOM. They never unmount. This preserves size state across collapse/expand cycles.
- `collapsedSize={0}` makes collapsed panels take zero space.
- `collapsible` enables the imperative `collapse()`/`expand()` methods.
- `panel.expand()` restores to the LAST known size (what the user dragged to), NOT `defaultSize`. This satisfies the locked decision "reopening a collapsed panel restores its last dragged width."
- `autoSaveId="void-main-layout"` persists panel sizes to localStorage automatically.
- The `panelAnimated` class is conditionally applied: present when NOT dragging (so collapse/expand animates), removed during drag (so pointer tracking is instant).
- `order` props ensure correct layout ordering.
- The center Panel has no `defaultSize` — it takes remaining space automatically.
- `minSize={10}` and `maxSize={40}` are percentages, roughly corresponding to the locked constraint "min 150px, max 50% viewport" on a typical ~1400px window.
